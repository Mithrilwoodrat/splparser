
Below are clips from an interactive session in which I test various aspects of the parser.

>>> import splparser.parser

>>> splparser.parser.parse("sort num(ip), -str(url)").print_tree()
('ROOT')
    ('STAGE')
        ('SORT')
            ('ASCENDING')
                ('NUM')
                    ('WORD': 'IP')
            ('DESCENDING')
                ('STR')
                    ('WORD': 'url')

>>> splparser.parser.parse("sort 100 +num(size), -str(url)").print_tree()
('ROOT')
    ('STAGE')
        ('SORT')
            ('INT': 100)
            ('ASCENDING')
                ('NUM')
                    ('WORD': 'size')
            ('DESCENDING')
                ('STR')
                    ('WORD': 'url')

>>> spl.parser.parser.parse("sort _time, -host").print_tree()
('ROOT')
    ('STAGE')
        ('SORT')
            ('ASCENDING')
                ('ID': '_time')
            ('DESCENDING')
                ('WORD': 'host')

>>> spl.parser.parser.parse("sort - Time").print_tree()
('ROOT')
    ('STAGE')
        ('SORT')
            ('DESCENDING')
                ('WORD': 'Time')

>>> spl.parser.parser.parse('sort + "Vegeta what does the scouter say about his power level"').print_tree()
('ROOT')
    ('STAGE')
        ('SORT')
            ('ASCENDING')
                ('ID': 'Vegeta what does the scouter say about his power level')

>>> spl.parser.parser.parse("sort limit=9001 +power(level)")
('ROOT')
    ('STAGE')
        ('SORT')
            ('EQ')
                ('WORD': 'limit')
                ('INT': '9001')
            ('ASCENDING')
                ('POWER')
                    ('WORD': 'level')
