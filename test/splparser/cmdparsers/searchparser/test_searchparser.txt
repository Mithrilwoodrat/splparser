
Below are clips from an interactive session in which I test various aspects of the parser.

>>> import splparser.parser

>>> parsed = splparser.parser.parse("search foo")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('WORD': 'foo')

>>> parsed = splparser.parser.parse("search foo | search bar")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('WORD': 'foo')
    ('STAGE')
        ('SEARCH')
            ('WORD': 'bar')
 
>>> parsed = splparser.parser.parse("search 123")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('INT': '123')

>>> parsed = splparser.parser.parse("search 12.34.12.34")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('IPV4ADDR': '12.34.12.34')

>>> parsed = splparser.parser.parse('search "literal with spaces"')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('LITERAL': '"literal with spaces"')

>>> parsed = splparser.parser.parse('search "word"')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('WORD': '"word"')

>>> parsed = splparser.parser.parse('search callme@maybe.com')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('EMAIL': 'callme@maybe.com')

>>> parsed = splparser.parser.parse('search 3.0')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('FLOAT': '3.0')

>>> parsed = splparser.parser.parse('search you AND this')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('AND')
                ('WORD': 'you')
                ('WORD': 'this')

>>> parsed = splparser.parser.parse('search just, met, you')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('AND')
                ('WORD': 'just')
                ('WORD': 'met')

>>> parsed = splparser.parser.parse('search all OR the AND other')

('ROOT')

        ('SEARCH')

                ('WORD': 'all')

>>> parsed = splparser.parser.parse('search host=value')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')

                ('HOST')
                ('WORD': 'value')

>>> parsed = splparser.parser.parse('search sourcetype=hey')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('EQ')
                ('SOURCETYPE')
                ('WORD': 'hey')

>>> parsed = splparser.parser.parse('search NOT "your mom"')
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('NOT')
                ('LITERAL': '"your mom"')
