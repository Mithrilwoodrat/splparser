
Below are clips from an interactive session in which I test various aspects of the parser.

>>> import splparser.parser

>>> splparser.parser.parse("search foo").print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('WORD': 'foo')

>>> splparser.parser.parse("search foo | search bar").print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('WORD': 'foo')
    ('STAGE')
        ('SEARCH')
            ('WORD': 'bar')
 
>>> splparser.parser.parse("search 123").print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('INT': '123')

>>> splparser.parser.parse("search 12.34.12.34").print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('IPV4ADDR': '12.34.12.34')

>>> splparser.parser.parse('search "literal with spaces"').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('LITERAL': '"literal with spaces"')

>>> splparser.parser.parse('search "word"').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('WORD': '"word"')

>>> splparser.parser.parse('search callme@maybe.com').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('EMAIL': 'callme@maybe.com')

>>> splparser.parser.parse('search 3.0').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('FLOAT': '3.0')

>>> splparser.parser.parse('search you AND this').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('AND')
                ('WORD': 'you')
                ('WORD': 'this')

>>> splparser.parser.parse('search just, met, you').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('AND')
                ('WORD': 'just')
                ('WORD': 'met')
                ('WORD': 'you')

>>> splparser.parser.parse('search all OR the AND other').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('OR')
                ('WORD': 'all')
                ('AND')
                    ('WORD': 'the')
                    ('WORD': 'other')

>>> splparser.parser.parse('search host=value').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('EQ')
                ('HOST')
                ('WORD': 'value')

>>> splparser.parser.parse('search sourcetype=hey').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('EQ')
                ('SOURCETYPE')
                ('WORD': 'hey')

>>> splparser.parser.parse('search NOT "your mom"').print_tree()
('ROOT')
    ('STAGE')
        ('SEARCH')
            ('NOT')
                ('LITERAL': '"your mom"')
