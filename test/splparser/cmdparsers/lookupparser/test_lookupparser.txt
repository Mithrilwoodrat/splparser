
Below are clips from an interactive session in which I test various aspects of the parser.

>>> import splparser.parser

>>> parsed = splparser.parser.parse("lookup sourcenames source")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('LOOKUP')
            ('WORD': 'sourcenames')
            ('WORD': 'source')

>>> parsed = splparser.parser.parse("lookup status_desc status OUTPUT description")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('LOOKUP')
            ('WORD': 'status_desc')
            ('WORD': 'status')
            ('OUTPUT')
                ('WORD': 'description')

>>> parsed = splparser.parser.parse("lookup local=true geoip clientip")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('LOOKUP')
            ('LOCAL')
                ('WORD': 'true')
            ('WORD': 'geoip')
            ('WORD': 'clientip')

>>> parsed = splparser.parser.parse("lookup usertogroup user as local_user OUTPUT group as user_group")
>>> parsed.print_tree()
('ROOT')
    ('STAGE')
        ('LOOKUP')
            ('WORD': 'usertogroup')
            ('WORD': 'user')
                ('AS')
                    ('WORD': 'local_user')
            ('OUTPUT')
                ('WORD': 'group')
                    ('AS')
                        ('WORD': 'user_group')
