Below are clips from an interactive session in which I test various aspects of the parser.

>>> import splparser

>>> splparser.parse('regex _raw="(?=!\d)10.\d{1,3}\.\d{1,3}\.\d{1,3}(?!\d)"').print_tree()
('ROOT')
    ('STAGE')
        ('REGEX')
            ('EQ')
                ('ID': '_raw')
                ('REGULAR_EXPRESSION': '"(?=!\d)10.\d{1,3}\.\d{1,3}\.\d{1,3}(?!\d)"')

>>> splparser.parse('regex _raw="complicated|regex(?=expression)"').print_tree()
('ROOT')
    ('STAGE')
        ('REGEX')
            ('EQ')
                ('ID': '_raw')
                ('REGULAR_EXPRESSION': '"complicated|regex(?=expression)"')

>>> splparser.parse('regex "Site Category"="Online Storage"').print_tree()
('ROOT')
    ('STAGE')
        ('REGEX')
            ('EQ')
                ('LITERAL': '"Site Category"')
                ('REGULAR_EXPRESSION': '"Online Storage"')

>>> splparser.parse('regex exception!="^\["').print_tree()
('ROOT')
    ('STAGE')
        ('REGEX')
            ('NEQ')
                ('WORD': 'exception')
                ('REGULAR_EXPRESSION': '"^\["')

>>> splparser.parse('regex _raw !="the object is in use by another process"').print_tree()
('ROOT')
    ('STAGE')
        ('REGEX')
            ('NEQ')
                ('ID': '_raw')
                ('REGULAR_EXPRESSION': '"the object is in use by another process"')
