>>> import splparser

>>> splparser.parse("appendcols [search 404]").print_tree()
('ROOT')
    ('STAGE')
        ('APPENDCOLS')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('SEARCH')
                            ('INT': '404')

>>> splparser.parse("appendcols maxtime=100 [search test | sort test]").print_tree()
('ROOT')
    ('STAGE')
        ('APPENDCOLS')
            ('MAXTIME')
                ('INT': '100')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('SEARCH')
                            ('WORD': 'test')
                    ('STAGE')
                        ('SORT')
                            ('ASCENDING')
                                ('WORD': 'test')
