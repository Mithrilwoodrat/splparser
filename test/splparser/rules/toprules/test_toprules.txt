
Below are clips from an interactive session in which I test various aspects of the parser.

>>> import splparser.parser

>>> splparser.parse("top blockedUrl").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'top')
            ('FIELD': 'blockedUrl')

>>> splparser.parse("top limit=10000 msize,Drive_Type by host").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'top')
            ('EQ': 'assign')
                ('OPTION': 'limit')
                ('VALUE': '10000')
            ('FUNCTION': 'groupby')
                ('FIELD': 'msize')
                ('FIELD': 'Drive_Type')
                ('GROUPING_DEFAULT_FIELD': 'host')

>>> splparser.parse("top limit=0 q").print_tree()

        ('COMMAND': 'top')

            ('FIELD': 'q')
