
Below are clips from an interactive session in which I test various aspects of the parser.

>>> import splparser.parser

>>> splparser.parse('replace *localhost with localhost in host').print_tree()
('ROOT')
    ('STAGE')
        ('REPLACE')
            ('WITH')
                ('ID': '*localhost')
                ('WORD': 'localhost')
            ('IN')
                ('WORD': 'host')

>>> splparser.parse('replace "* localhost" with "localhost *" in host').print_tree()
('ROOT')
    ('STAGE')
        ('REPLACE')
            ('WITH')
                ('LITERAL': '"* localhost"')
                ('LITERAL': '"localhost *"')
            ('IN')
                ('WORD': 'host')

>>> splparser.parse('replace aug with August in start_month end_month').print_tree()
('ROOT')
    ('STAGE')
        ('REPLACE')
            ('WITH')
                ('WORD': 'aug')
                ('WORD': 'August')
            ('IN')
                ('ID': 'start_month')
                ('ID': 'end_month')
