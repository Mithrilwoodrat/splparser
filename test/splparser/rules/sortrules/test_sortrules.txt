>>> import splparser

>>> splparser.parse("sort num(ip), -str(url)").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'sort')
            ('FUNCTION': 'ascending')
                ('FUNCTION': 'num')
                    ('FIELD': 'ip')
            ('FUNCTION': 'descending')
                ('FUNCTION': 'str')
                    ('FIELD': 'url')

>>> splparser.parse("sort 100 +num(size), -str(url)").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'sort')
            ('EQ': 'assign')
                ('OPTION')
                ('VALUE': '100')
            ('FUNCTION': 'ascending')
                ('FUNCTION': 'num')
                    ('FIELD': 'size')
            ('FUNCTION': 'descending')
                ('FUNCTION': 'str')
                    ('FIELD': 'url')

>>> splparser.parse("sort _time, -host").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'sort')
            ('FUNCTION': 'ascending')
                ('INTERNAL_FIELD': '_time')
            ('FUNCTION': 'descending')
                ('DEFAULT_FIELD': 'host')
