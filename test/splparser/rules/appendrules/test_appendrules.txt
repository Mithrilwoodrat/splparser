>>> import splparser

>>> splparser.parse("append [search foo]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'append')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('VALUE': 'foo')

>>> splparser.parse("append maxtime=100 [search test | sort test]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'append')
            ('EQ')
                ('OPTION': 'maxtime')
                ('INT': '100')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('WORD': 'test')
                    ('STAGE')
                        ('COMMAND': 'sort')
                            ('FUNCTION': 'ascending')
                                ('WORD': 'test')

>>> splparser.parse("append maxtime=100 maxout=50 timeout=10 [search test | sort test | top 5 lol by rofl]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'append')
            ('EQ')
                ('OPTION': 'maxtime')
                ('VALUE': '100')
            ('EQ')
                ('OPTION': 'maxout')
                ('VALUE': '50')
            ('EQ')
                ('OPTION': 'timeout')
                ('VALUE': '10')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('VALUE': 'test')
                    ('STAGE')
                        ('COMMAND': 'sort')
                            ('FUNCTION': 'ascending')
                                ('FIELD': 'test')
                    ('STAGE')
                        ('COMMAND': 'top')
                            ('VALUE': '5')
                            ('FIELD': 'lol')
                            ('BY')
                                ('FIELD': 'rofl')

>>> splparser.parse("append [search foo | append [search foo]]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'append')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('VALUE': 'foo')
                    ('STAGE')
                        ('COMMAND': 'append')
                            ('SUBSEARCH')
                                ('ROOT')
                                    ('STAGE')
                                        ('COMMAND': 'search')
                                            ('VALUE': 'foo')
