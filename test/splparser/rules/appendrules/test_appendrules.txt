>>> import splparser

>>> splparser.parse("append [search foo]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'append')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('FUNCTION': 'contains')
                                ('VALUE': 'foo')

>>> splparser.parse("append maxtime=100 [search test | sort test]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'append')
            ('EQ': 'assign')
                ('OPTION': 'maxtime')
                ('VALUE': '100')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('FUNCTION': 'contains')
                                ('VALUE': 'test')
                    ('STAGE')
                        ('COMMAND': 'sort')
                            ('FUNCTION': 'ascending')
                                ('FIELD': 'test')

>>> splparser.parse("append maxtime=100 maxout=50 timeout=10 [search test | sort test | top lol by rofl]").print_tree()

                ('OPTION': 'timeout')
                ('VALUE': '10')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('FUNCTION': 'contains')
                                ('VALUE': 'test')
                    ('STAGE')
                        ('COMMAND': 'sort')
                            ('FUNCTION': 'ascending')
                                ('FIELD': 'test')
                    ('STAGE')
                        ('COMMAND': 'top')
                            ('FUNCTION': 'groupby')
