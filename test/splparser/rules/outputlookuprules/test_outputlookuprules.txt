>>> import splparser

>>> splparser.parse("outputlookup tablename").print_tree()
('ROOT')
    ('STAGE')
        ('OUTPUTLOOKUP')
            ('WORD': 'tablename')

>>> splparser.parse("outputlookup append=true tablename").print_tree()
('ROOT')
    ('STAGE')
        ('OUTPUTLOOKUP')
            ('EQ')
                ('WORD': 'append')
                ('WORD': 'true')
            ('WORD': 'tablename')
