>>> import splparser

>>> splparser.parse("inputlookup tablename")
('ROOT')
    ('STAGE')
        ('INPUTLOOKUP')
            ('WORD': 'filename')

>>> splparser.parse("inputlookup append=true tablename")
('ROOT')
    ('STAGE')
        ('INPUTLOOKUP')
            ('EQ')
                ('WORD': 'append')
                ('WORD': 'true')
            ('WORD': 'tablename')

>>> splparser.parse("inputlookup append=t userlistbyclientip.csv")
('ROOT')
    ('STAGE')
        ('INPUTLOOKUP')
            ('EQ')
                ('WORD': 'append')
                ('WORD': 't')
            ('ID': 'userlistbyclientip.csv')

>>> splparser.parse("inputlookup append=t start=100 max=1337 userlistbyclientip.csv")
('ROOT')
    ('STAGE')
        ('INPUTLOOKUP')
            ('EQ')
                ('WORD': 'append')
                ('WORD': 't')
            ('EQ')
                ('WORD': 'start')
                ('INT': '100')
            ('EQ')
                ('WORD': 'max')
                ('INT': '1337')
            ('ID': 'userlistbyclientip.csv')
