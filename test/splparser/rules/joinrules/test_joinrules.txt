>>> import splparser

>>> splparser.parse("join [search foo]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'join')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('FUNCTION': 'contains')
                                ('VALUE': 'foo')

>>> splparser.parse("join type=inner [search test | sort test]").print_tree()
('ROOT')
    ('STAGE')
        ('COMMAND': 'join')
            ('EQ': 'assign')
                ('OPTION': 'type')
                ('VALUE': 'inner')
            ('SUBSEARCH')
                ('ROOT')
                    ('STAGE')
                        ('COMMAND': 'search')
                            ('FUNCTION': 'contains')
                                ('VALUE': 'test')
                    ('STAGE')
                        ('COMMAND': 'sort')
                            ('FUNCTION': 'ascending')
                                ('FIELD': 'test')
